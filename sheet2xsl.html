'将工作簿所有工作表另存为单独的文件。 '路径为原工作簿路径，文件名为工作表名 

Sub Test() 
    Dim Sht As Worksheet 
    For Each Sht In Sheets 
        Sht.Copy 
        ActiveWorkbook.SaveAs ThisWorkbook.Path & "\" & Sht.Name & ".xls" 
        ActiveWorkbook.Close 
    Next 
End Sub
